leetcode 17. 电话号码的字母组合

2020-08-27

思路

爆搜就完事了。。。

class Solution {
public:
    map<int, vector<char>> m;
    vector<string> ans;
    string temp;
    void dfs(string digits, int step) {
        if (step >= digits.size()) {
            ans.push_back(temp);
            return;
        }
        int num = digits[step] - '0';
        auto letters = m[num];
        for (auto ch : letters) {
            temp.push_back(ch);
            dfs(digits, step + 1);
            temp.pop_back();
        }
    }
    vector<string> letterCombinations(string digits) {
        if (digits.size() == 0) return {};
        m[2] = {'a', 'b', 'c'};
        m[3] = {'d', 'e', 'f'};
        m[4] = {'g', 'h', 'i'};
        m[5] = {'j', 'k', 'l'};
        m[6] = {'m', 'n', 'o'};
        m[7] = {'p','q','r', 's'};
        m[8] = {'t', 'u', 'v'};
        m[9] = {'w', 'x', 'y', 'z'};
        ans = {};
        temp = {};
        dfs(digits, 0);
        return ans;
    }
};
