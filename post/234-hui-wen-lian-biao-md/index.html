leetcode 234. 回文链表

2020-10-23 leetcode

思路

1.求出中间结点 以分割整条链表

2.从中间结点开始 翻转后半部分链表

3.从两端开始比较

/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func middleList(head *ListNode) *ListNode {
    slow, fast := head, head
    for fast.Next != nil && fast.Next.Next != nil{
        slow = slow.Next;
        fast = fast.Next.Next;
    }
    return slow
}
func reverseList(head *ListNode) *ListNode {
    cur := head
    var prev *ListNode 
    prev = nil 
    for cur != nil {
        tmpNext := cur.Next
        //1->2->3 1<-2
        cur.Next = prev
        prev = cur
        cur = tmpNext
    }
    return prev
}
func isPalindrome(head *ListNode) bool {
    if head == nil {
        return true
    }
    s1 := head
    s2 := middleList(head)
    s2 = reverseList(s2.Next)
    for s1 != nil && s2 != nil {
        if s1.Val != s2.Val {
            return false
        }
        fmt.Println(s1.Val, s2.Val)
        s1 = s1.Next
        s2 = s2.Next
    }
    return true
}
