leetcode 130. 被围绕的区域

2020-08-11

思路

思路很简单我们只要从边界部分做dfs或者bfs就好并将搜索联通块标记为‘#’即可。最后遍历一遍棋盘所有未被标记成‘#’的‘O’更新为‘X’，再将‘#’更新为'O'就好。

class Solution {
public:
    int n, m;
    void dfs(int x, int y, vector<vector<char>>& board) {
        if (x < 0 || x >= n || y < 0 || y >= m || board[x][y] != 'O') return;

        board[x][y] = '#';
        dfs(x-1,y, board);
        dfs(x+1,y, board);
        dfs(x, y-1, board);
        dfs(x, y+1, board);
    }
    void solve(vector<vector<char>>& board) {
        n = board.size();
        if (n == 0) return;
        m = board[0].size();
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                bool flag = (i == 0 || j == 0 || i == n - 1 || j == m - 1);
                if (flag && 'O' == board[i][j]) {
                    dfs(i, j, board);
                }
            }
        }
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (board[i][j] == '#'){
                    board[i][j] = 'O';
                } else if (board[i][j] == 'O'){
                    board[i][j] = 'X';
                }
            }
        }
    }
};
