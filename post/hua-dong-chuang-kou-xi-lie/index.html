滑动窗口系列

2020-08-10 leetcode

思路

1、我们在字符串 S 中使用双指针中的左右指针技巧，初始化 left = right = 0，把索引左闭右开区间 [left, right) 称为一个「窗口」。

2、我们先不断地增加 right 指针扩大窗口 [left, right)，直到窗口中的字符串符合要求（包含了 T 中的所有字符）。

3、此时，我们停止增加 right，转而不断增加 left 指针缩小窗口 [left, right)，直到窗口中的字符串不再符合要求（不包含 T 中的所有字符了）。同时，每次增加 left，我们都要更新一轮结果。

4、重复第 2 和第 3 步，直到 right 到达字符串 S 的尽头。

这个思路其实也不难，**第 2 步相当于在寻找一个「可行解」，然后第 3 步在优化这个「可行解」，最终找到最优解，**也就是最短的覆盖子串。左右指针轮流前进，窗口大小增增减减，窗口不断向右滑动，这就是「滑动窗口」这个名字的来历。

Leetcode 438. 找到字符串中所有字母异位词

class Solution {
public:
    vector<int> findAnagrams(string s, string p) {
        unordered_map<char, int> window, need;
        for (char ch : p) {
            need[ch]++;
        }
        int left = 0, right = 0;
        int cnt = 0;
        vector<int> res;
        while(right < s.size()) {
            char c = s[right];
            right++;
            if (need.count(c)) {
                window[c]++;
                if (window[c] == need[c]) {//更新的是窗口里的字符数量 和 need中作对比
                    cnt++;
                }
            }
            while (right - left >= p.size()) {
                if (cnt == need.size()) {
                    res.push_back(left);
                }
                char del = s[left];
                left++;
                if (need.count(del)){
                    if (window[del] == need[del]) cnt--;
                    window[del]--;
                }
            }
        }
        return res;

    }
};

LeetCode 76. 最小覆盖子串


func minWindow(s string, t string) string {
    window := make(map[byte]int)
    need := make(map[byte]int)
    // Len := 0
    for i := range t {
        need[t[i]]++
    }

    left, right, valid := 0, 0, 0
    Len := math.MaxInt32
    index := 0
    for right < len(s) {
        c := s[right]
        right++
        if _, ok := need[c]; ok {
            window[c]++
            if window[c] == need[c] {
                valid++
            }
        }
        for valid == len(need) {//表示所有字符以找到，答案就在这个范围内，开始缩小窗口
            if right - left < Len {
                Len = right - left
                index = left
            }
            del := s[left]
            left++
            if _, ok := need[del]; ok{
                if window[del] == need[del] {
                    valid--;
                }
                window[del]--;
            }
        }
    }
    if Len == math.MaxInt32 {
        return ""
    }
    return s[index:index+Len]
}

LeetCode 3. 无重复字符的最长子串

class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        unordered_map<char, int> window;
        int left = 0, right = 0, Len = INT_MIN;
        while (right < s.size()) {//永远记住边界是[left, right)
            char add = s[right];
            right++;
            window[add]++;
            while(window[add] > 1) {
                window[s[left]]--;
                left++;
            }
            Len = max(Len, right - left);
        }
        return Len == INT_MIN ? 0 : Len;
    }
};

209. 长度最小的子数组

class Solution {
public:
    int minSubArrayLen(int s, vector<int>& nums) {
        int l = 0, r = 0;
        int sum = 0;
        int len = INT_MAX;
        while(r < nums.size()) {
            sum += nums[r];
            r++;
            while (sum >= s) {
                len = min(len, r - l);
                sum -= nums[l];
                l++;
            }
        }
        return len == INT_MAX ? 0 : len;
    }
};
