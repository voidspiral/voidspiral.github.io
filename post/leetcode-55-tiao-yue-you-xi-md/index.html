leetcode 55. 跳跃游戏

2020-10-23 1 min read # leetcode

思路

贪心就完事了 考虑能走的最远距离可以到达边界就可以直接返回了

class Solution {
public:
    bool canJump(vector<int>& nums) {
       int movemost = 0;
       for (int i = 0; i < nums.size(); i++) {
           if (i <= movemost) {
               movemost = max(movemost, i + nums[i]);
               if (movemost >= nums.size() - 1){
                   return true;
               }
           }
       }
       return false;
    }
};
