04 1553. 吃掉 N 个橘子的最少天数

思路1

bfs 记忆化 因为是层序遍历 所以一定可以保证答案是路径最短的 也就是步骤最少的结果

class Solution {
public:
    unordered_map<int, int> used;
    // queue<int> que;
    // void check(int d, int x) {
    //     if (used.count(x)) return;
    //     use[x] = d + 1;
    //     que.push(x);
    // }
    int minDays(int n) {
        queue<int> que;
        que.push(n);
        int step = 0;
        used[n] = 1;
        while(que.size()){
            int cur = que.front();
            if (cur == 1) return used[cur];
            que.pop();
            if (used.count(cur - 1) == 0) {
                used[cur-1] = used[cur] + 1;
                que.push(cur-1);
            }
            if (cur % 2 == 0 && used.count(cur/2) == 0) {
                used[cur/2] = used[cur] + 1;
                que.push(cur/2);
            }
            if (cur % 3 == 0 && used.count(cur/3) == 0) {
                used[cur/3] = used[cur] + 1;
                que.push(cur/3);
            }
        }
        return -1;
    }
};

思路2

DFS 记忆化 + 剪枝 详细思路可以参考[1]、[2]

class Solution {
public:
    unordered_map<int, int> dp;
    int dfs(int n) {
        if (n == 1) return 1;
        if (n == 2 || n == 3) return 2;
        // if (dp.count(n)) return dp[n];
        auto it = dp.find(n);
        if (it != dp.end()) return it->second;
        return dp[n] = min(dfs(n/2) + (n % 2) + 1, dfs(n/3) + (n % 3) + 1);
    }
    int minDays(int n) {
        // dp.clear();
        return dfs(n);
    }
};

03 1552. 两球之间的磁力

class Solution {
public:
    bool check(vector<int>& pos, int len, int m) {
        int cnt = 1;
        int t = pos[0];
        for (int i = 1; i < pos.size(); i++) {
            if (pos[i] - t >= len) {
                t = pos[i];
                cnt++;
                if (cnt >= m) break;
            }
        }
        return cnt >= m;
    }
    int maxDistance(vector<int>& position, int m) {
        sort(position.begin(), position.end());
        int l = 1, r = position[position.size()-1];
        int ans = -1;
        while(l <= r){
            int mid = (l + r) / 2;
            if (check(position, mid, m)) {
                l = mid + 1;
                ans = mid;
            } else {
                r = mid - 1;
            }
        }
        return ans;
    }
};

02 1551. 使数组中所有元素相等的最小操作数

class Solution {
public:
    int minOperations(int n) {
        vector<int> nums(n);
        int sum = 0;
        for (int i = 0; i < n; i++) {
            nums[i] = 2 * i + 1;
            sum += nums[i];
        }
        int target = sum / n;
        int ans = 0;
        for (int i = 0; i < n ; i++) {
            if (target < nums[i]) ans += abs(target-nums[i]);
            // cout << ans << endl;
        }
            
        return ans;
    }
};

01 1550. 存在连续三个奇数的数组

class Solution {
public:
    bool threeConsecutiveOdds(vector<int>& arr) {
        if (arr.size() < 3) return false;
        for (int i = 0; i < arr.size() - 2; i++) {
            if ((arr[i] & 1) && (arr[i+1] & 1) && (arr[i+2] & 1)) {
                return true;
            } 
        }
        return false;
    }
};
