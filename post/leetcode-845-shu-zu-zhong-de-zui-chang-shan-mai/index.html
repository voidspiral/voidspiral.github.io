leetcode 845. 数组中的最长山脉

2020-10-27

思路

1.类似最长上升子序列 我们先从左到右求一遍 再从右到左求一遍

func longestMountain(A []int) int {
    if len(A) < 3 {
        return 0
    }
    left := make([]int, len(A))
    n := len(A)
    for i := 1; i < n; i ++ {
        if A[i-1] < A[i] {
           left[i] = left[i - 1] + 1 
        }
    }

    right := make([]int, len(A))
    for i := n - 2; i >= 0; i-- {
        if (A[i+1] < A[i]) {
            right[i] = right[i + 1] + 1
        }
    }
    ans := 0
    for i := 0; i < len(A); i++ {
        if left[i] > 0 && right[i] > 0 {
            if ans < left[i] + right[i] + 1{
                ans = left[i] + right[i] + 1
            }
        }
    }
    return ans
}
